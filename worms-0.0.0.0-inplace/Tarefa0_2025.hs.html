   15 -- | Retorna a quantidade de munições disponíveis de uma minhoca para uma dada arma.
   16 encontraQuantidadeArmaMinhoca :: TipoArma -> Minhoca -> Int
   17 encontraQuantidadeArmaMinhoca tipo m = case tipo of
   18     Jetpack -> jetpackMinhoca m
   19     Escavadora -> escavadoraMinhoca m
   20     Bazuca -> bazucaMinhoca m
   21     Mina -> minaMinhoca m
   22     Dinamite -> dinamiteMinhoca m
   23 
   24 -- | Atualia a quantidade de munições disponíveis de uma minhoca para uma dada arma.
   25 atualizaQuantidadeArmaMinhoca :: TipoArma -> Minhoca -> Int -> Minhoca
   26 atualizaQuantidadeArmaMinhoca tipo m i = case tipo of
   27     Jetpack -> m {jetpackMinhoca = jetpackMinhoca m + i}
   28     Escavadora -> m {escavadoraMinhoca = escavadoraMinhoca m + i}
   29     Bazuca -> m {bazucaMinhoca = bazucaMinhoca m + i}
   30     Mina -> m {minaMinhoca = minaMinhoca m + i}
   31     Dinamite -> m {dinamiteMinhoca = dinamiteMinhoca m + i}

   35 -- | Verifica se um tipo de terreno é destrutível, i.e., pode ser destruído por explosões.
   36 --
   37 -- __NB:__ Apenas @Terra@ é destrutível.
   38 eTerrenoDestrutivel :: Terreno -> Bool
   39 eTerrenoDestrutivel t = if t == Terra then True else False

   41 -- | Verifica se um tipo de terreno é opaco, i.e., não permite que objetos ou minhocas se encontrem por cima dele.
   42 --
   43 -- __NB:__ Apenas @Terra@ ou @Pedra@ são opacos.
   44 eTerrenoOpaco :: Terreno -> Bool
   45 eTerrenoOpaco t = if t == Pedra || t == Terra then True else False

   47 -- | Verifica se uma posição do mapa está livre, i.e., pode ser ocupada por um objeto ou minhoca.
   48 --  
   49 -- __NB:__ Uma posição está livre se não contiver um terreno opaco.
   50 ePosicaoMapaLivre :: Posicao -> Mapa -> Bool
   51 ePosicaoMapaLivre (x,y) [] = False
   52 ePosicaoMapaLivre (0,y) (h:t) = encontraLinhaMapa y h
   53 ePosicaoMapaLivre (x,y) (h:t) = ePosicaoMapaLivre (x-1,y) t
   54

   56 encontraLinhaMapa :: Int -> [Terreno] -> Bool
   57 encontraLinhaMapa i [] = False
   58 encontraLinhaMapa 0 (h:t) = not (eTerrenoOpaco h)
   59 encontraLinhaMapa i (h:t) = encontraLinhaMapa (i-1) t

   64 -- | Verifica se uma posição do estado está livre, i.e., pode ser ocupada por um objeto ou minhoca.
   65 --
   66 -- __NB:__ Uma posição está livre se o mapa estiver livre e se não estiver já uma minhoca ou um barril nessa posição.
   67 ePosicaoEstadoLivre :: Posicao -> Estado -> Bool
   68 ePosicaoEstadoLivre pos estado =
   69     ePosicaoMapaLivre pos mapa && not (existeMinhoca pos minhocas) && not (existeObjeto pos objetos)
   70   where
   71     mapa = mapaEstado estado
   72     objetos = objetosEstado estado
   73     minhocas = minhocasEstado estado

   79 existeMinhoca :: Posicao -> [Minhoca] -> Bool
   80 existeMinhoca pos [] = False
   81 existeMinhoca (x,y) (h:t) = let pos = posicaoMinhoca h
   82                             in if pos == Just (x,y) then True else existeMinhoca (x,y) t
   83                        
   84 existeObjeto :: Posicao -> [Objeto] -> Bool
   85 existeObjeto pos [] = False
   86 existeObjeto pos (h:t) = let p = posicaoObjeto h
   87                          in if p == pos then True else existeObjeto pos t

   92 posicaoObjeto d@(Disparo {})  = posicaoDisparo d

   99 -- | Verifica se numa lista de objetos já existe um disparo feito para uma dada arma por uma dada minhoca.
  100 minhocaTemDisparo :: TipoArma -> NumMinhoca -> [Objeto] -> Bool
  101 minhocaTemDisparo _ _ [] = False 
  102 minhocaTemDisparo tipo indiceMinhoca (h:ls) = if (i == indiceMinhoca && t == tipo) then True else minhocaTemDisparo tipo indiceMinhoca ls
  103     where
  104         i = donoDisparo h
  105         t = tipoDisparo h
