   16 -- | Uma matriz é um conjunto de elementos a duas dimensões.
   17 --
   18 -- Em notação matemática, é geralmente representada por:
   19 --
   20 -- <<https://haslab.github.io/Teaching/LI1/2526/img/matriz.png>>
   21 type Matriz a = [[a]]
   22 
   23 -- | Uma posição numa matriz é dada como um par (/linha/,/colunha/).
   24 -- As coordenadas são dois números naturais e começam com (0,0) no canto superior esquerdo, com as linhas incrementando para baixo e as colunas incrementando para a direita.
   25 --
   26 -- <<https://haslab.github.io/Teaching/LI1/2526/img/posicaomatriz.png>>
   27 type Posicao = (Int,Int)
   28 
   29 -- | A dimensão de uma matrix dada como um par (/número de linhas/,/número de colunhas/).
   30 type Dimensao = (Int,Int)
   31 
   32 -- | Uma direção é dada pela rosa dos ventos. Ou seja, os 4 pontos cardeais e os 4 pontos colaterais.
   33 --
   34 -- <<https://haslab.github.io/Teaching/LI1/2526/img/rosadosventos.jpg>>
   35 data Direcao = Norte | Nordeste | Este | Sudeste | Sul | Sudoeste | Oeste | Noroeste
   36     deriving (Eq,Ord,Show,Read,Enum)

   40 -- | Verifica se o indice pertence à lista.
   41 eIndiceListaValido :: Int -> [a] -> Bool
   42 eIndiceListaValido x a = (x <= length(a)-1) && (x>(-1))
   43 
   44 -- | Calcula a dimensão de uma matriz.
   45 --
   46 -- __NB:__ Note que não existem matrizes de dimensão /m * 0/ ou /0 * n/, e que qualquer matriz vazia deve ter dimensão /0 * 0/.
   47 dimensaoMatriz :: Matriz a -> Dimensao
   48 dimensaoMatriz a = (length a - 1, length(head a) - 1)
   49 
   50 -- | Verifica se a posição pertence à matriz.
   51 ePosicaoMatrizValida :: Posicao -> Matriz a -> Bool 
   52 ePosicaoMatrizValida (a,b) m = a <= length m - 1 && b <= length(head m) - 1 && a >= 0 && b >= 0

   54 -- | Move uma posição uma unidade no sentido de uma direção.
   55 movePosicao :: Direcao -> Posicao -> Posicao
   56 movePosicao d (x, y) = case d of
   57     Norte     -> (x - 1, y)
   58     Sul       -> (x + 1, y)
   59     Este      -> (x, y + 1)
   60     Oeste     -> (x, y - 1)
   61     Nordeste  -> (x - 1, y + 1)
   62     Noroeste  -> (x - 1, y - 1)
   63     Sudeste   -> (x + 1, y + 1)
   64     Sudoeste  -> (x + 1, y - 1)
   65

   82 -- | Versão da função 'movePosicao' que garante que o movimento não se desloca para fora de uma janela.
   83 --
   84 -- __NB:__ Considere uma janela retangular com origem no canto superior esquerdo definida como uma matriz. A função recebe a dimensao da janela.
   85 movePosicaoJanela :: Dimensao -> Direcao -> Posicao -> Posicao
   86 movePosicaoJanela (dy,dx) dir pos = 
   87     if (yf < dy) && (xf < dx) && (yf >= 0) && (xf >= 0)
   88         then (xf, yf)
   89         else pos
   90     where
   91         (yf, xf) = movePosicao dir pos

   95 -- | Converte uma posição no referencial em que a origem é no canto superior esquerdo da janela numa posição em que a origem passa a estar no centro da janela.
   96 --
   97 -- __NB:__ Considere posições válidas. Efetue arredondamentos como achar necessário.
   98 origemAoCentro :: Dimensao -> Posicao -> Posicao
   99 origemAoCentro (dy, dx) (y, x) = (y - div dy 2, x - div dx 2)

  103 -- | Roda um par (posição,direção) 45% para a direita.
  104 --
  105 -- __NB:__ Vendo um par (posição,direção) como um vector, cria um novo vetor do desto com a próxima direção da rosa dos ventos rodando para a direita.
  106 --
  107 -- <<https://haslab.github.io/Teaching/LI1/2526/img/rodaposicaodirecao.png>>
  108 
  109 rodaPosicaoDirecao :: (Posicao,Direcao) -> (Posicao,Direcao)
  110 rodaPosicaoDirecao (pos, d) = (p, novaDirecao) -- Retorna a tupla (Posicao, Direcao)
  111     where 
  112         novaDirecao = case d of
  113             Norte    -> Nordeste
  114             Sul      -> Sudoeste
  115             Este     -> Sudeste
  116             Oeste    -> Noroeste
  117             Nordeste -> Este
  118             Sudeste  -> Sul
  119             Noroeste -> Norte
  120             Sudoeste -> Oeste
  121         
  122         p = movePosicao novaDirecao pos

  126 -- | Devolve o elemento num dado índice de uma lista.
  127 --
  128 -- __NB:__ Retorna @Nothing@ se o índice não existir.
  129 encontraIndiceLista :: Int -> [a] -> Maybe a
  130 encontraIndiceLista x [] = Nothing
  131 encontraIndiceLista x (h:t) = 
  132     if x < 0 then Nothing else if x == 0 then Just h else encontraIndiceLista(x-1) t
  133 
  134 
  135 -- | Modifica um elemento num dado índice.
  136 --
  137 -- __NB:__ Devolve a própria lista se o elemento não existir.
  138 atualizaIndiceLista :: Int -> a -> [a] -> [a]
  139 atualizaIndiceLista i n [] = []
  140 atualizaIndiceLista 0 n (h:t) = n : t
  141 atualizaIndiceLista i n (h:t)= h : atualizaIndiceLista (i-1) n t

  145 -- | Devolve o elemento numa dada posição de uma matriz.
  146 --
  147 -- __NB:__ Retorna @Nothing@ se a posição não existir.
  148 encontraPosicaoMatriz :: Posicao -> Matriz a -> Maybe a
  149 encontraPosicaoMatriz pos [] = Nothing
  150 encontraPosicaoMatriz (0,c) (h:t) = encontraLista c h
  151 encontraPosicaoMatriz (l,c) (h:t) = encontraPosicaoMatriz (l-1,c) t
  152 
  153 
  154 encontraLista :: Int -> [a] -> Maybe a
  155 encontraLista i [] = Nothing
  156 encontraLista i (h:t) |i == 0 = Just h 
  157                       |otherwise = encontraLista (i-1) t
  158 
  159 
  160 
  161 -- | Modifica um elemento numa dada posição de uma matriz.
  162 --
  163 -- __NB:__ Devolve a própria matriz se o elemento não existir.
  164 
  165 atualizaPosicaoMatriz :: Posicao -> a -> Matriz a -> Matriz a
  166 atualizaPosicaoMatriz _ _ [] = []
  167 atualizaPosicaoMatriz (0,c) x (h:t) = substituirLista c x h : t
  168 atualizaPosicaoMatriz (l,c) x (h:t) = h : atualizaPosicaoMatriz (l - 1, c) x t
  169 
  170 substituirLista :: Int -> a -> [a] -> [a]
  171 substituirLista _ _ [] = []
  172 substituirLista 0 x (_:t) = x : t
  173 substituirLista i x (h:t) = h : substituirLista (i - 1) x t

  180 -- | Aplica uma sequência de movimentações a uma posição, pela ordem em que ocorrem na lista.
  181 moveDirecoesPosicao :: [Direcao] -> Posicao -> Posicao
  182 moveDirecoesPosicao [] pos = pos
  183 moveDirecoesPosicao (h:t) pos = moveDirecoesPosicao t (movePosicao h pos)
  184 
  185 
  186 -- | Aplica a mesma movimentação a uma lista de posições.
  187 moveDirecaoPosicoes :: Direcao -> [Posicao] -> [Posicao]
  188 moveDirecaoPosicoes dir [] = []
  189 moveDirecaoPosicoes dir (h:t) = movePosicao dir h : moveDirecaoPosicoes dir t

  191 -- | Verifica se uma matriz é válida, no sentido em que modela um rectângulo.
  192 --
  193 -- __NB:__ Todas as linhas devem ter o mesmo número de colunas. 
  194 eMatrizValida :: Matriz a -> Bool
  195 eMatrizValida [] = False
  196 eMatrizValida [h] = True
  197 eMatrizValida (h1:h2:t) = length h1 == length h2 && eMatrizValida (h2:t)
