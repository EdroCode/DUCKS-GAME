   16 -- | Um tipo de terreno do mapa.
   17 data Terreno
   18     -- | Terreno vazio.
   19     = Ar
   20     -- | Terreno que afoga minhocas.
   21     | Agua
   22     -- | Terreno opaco e destrutivel.
   23     | Terra
   24     -- | Terreno opaco e indestrutivel.
   25     | Pedra
   26     deriving (Eq,Ord,Show,Read,Enum)
   27 
   28 -- | O mapa do jogo é uma matriz de terrenos.
   29 type Mapa = Matriz Terreno
   30 
   31 -- | Os diversos tipos de arma disponíveis para uma minhoca.
   32 data TipoArma = Jetpack | Escavadora | Bazuca | Mina | Dinamite
   33     deriving (Eq,Ord,Show,Read,Enum)
   34 
   35 -- | O estado de saúde de uma minhoca.
   36 data VidaMinhoca
   37     -- | Está viva com um número inteiros de pontos de vida.
   38     = Viva Int
   39     -- | Está morta.
   40     | Morta
   41     deriving (Eq,Ord,Show,Read)
   42 
   43 -- | O estado completo de uma minhoca.
   44 data Minhoca = Minhoca
   45     -- | Uma posição no mapa. Opcional porque a minhoca pode ter saído do mapa.
   46     { posicaoMinhoca :: Maybe Posicao
   47 
   48     , vidaMinhoca :: VidaMinhoca -- | O estado de saúde da minhoca.
   49     -- | Munições de @Jetpack@.
   50     , jetpackMinhoca :: Int
   51     -- | Munições de @Escavadora@.
   52     , escavadoraMinhoca :: Int
   53     -- | Munições de @Bazuca@.
   54     , bazucaMinhoca :: Int
   55     -- | Munições de @Mina@.
   56     , minaMinhoca :: Int
   57     -- | Munições de @Dinamite@.
   58     , dinamiteMinhoca :: Int
   59     }
   60     deriving (Eq,Ord,Show,Read)
   61 
   62 -- | Um tick é a unidade de tempo do jogo.
   63 type Ticks = Int
   64 
   65 -- | O índice de uma minhoca na lista de minhocas.
   66 type NumMinhoca = Int
   67 
   68 -- | O índice de um objeto na lista de objetos.
   69 type NumObjeto = Int
   70 
   71 -- | Um objeto colocado no mapa.
   72 data Objeto
   73     -- | Um disparo de uma arma.
   74     = Disparo
   75         -- | A posição do disparo no mapa.
   76         { posicaoDisparo :: Posicao
   77         -- | A direção do disparo.
   78         , direcaoDisparo :: Direcao
   79         -- | O tipo de arma do disparo.
   80         , tipoDisparo :: TipoArma
   81         -- | O tempo até o disparo explodir. Opcional porque nem todos os disparos de todas as armas têm um tempo pré-definido para explodir.
   82         , tempoDisparo :: Maybe Ticks
   83         -- | A minhoca que efetuou o disparo.
   84         , donoDisparo :: NumMinhoca
   85         }
   86     -- | Um barril de pólvora.
   87     | Barril
   88         -- | A posição do barril no mapa.
   89         { posicaoBarril :: Posicao
   90         -- | Se o barril está prestes a explodir ou não.
   91         , explodeBarril :: Bool
   92         }
   93     deriving (Eq,Ord,Show,Read)
   94 
   95 -- | Estado do jogo.
   96 data Estado = Estado
   97     -- | O mapa atual.
   98     { mapaEstado :: Mapa
   99     -- | Uma lista com os objetos presentes no mapa. Para as funções que vai desenvolver, deve considerar que a ordem dos elementos é irrelevante.
  100     , objetosEstado :: [Objeto]
  101     -- | Uma lista com as minhocas no jogo. A ordem dos elementos é relevante, no sentido cada minhoca vai ser identificada pelo seu índice na lista.
  102     , minhocasEstado :: [Minhoca]
  103     }
  104     deriving (Eq,Ord,Show,Read)
  105 
  106 -- | Uma jogada que uma minhoca pode efetuar.
  107 data Jogada
  108     -- | Disparar uma arma numa dada direção.
  109     = Dispara TipoArma Direcao
  110     -- | Mover-se numa dada direção.
  111     | Move Direcao
  112     deriving (Eq,Ord,Show,Read)
