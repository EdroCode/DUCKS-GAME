   17 -- | Função principa*! Excecao nl da Tarefa 1. Recebe um estado e retorna se este é válido ou não.
   18 validaEstado :: Estado -> Bool
   19 validaEstado e = eMapaValido mapa && eObjetosValido e objetos && eMinhocasValidas e minhocas
   20     where
   21         mapa = mapaEstado e
   22         objetos = objetosEstado e
   23         minhocas = minhocasEstado e
   24 
   25 -- * Valida se o mapa é valido

   27 eMapaValido :: Mapa -> Bool 
   28 eMapaValido [] = False    
   29 eMapaValido m = (eMatrizValida m)

   33 -- * Valida se o objeto é valido
   34 
   35 eObjetosValido :: Estado -> [Objeto] -> Bool
   36 eObjetosValido _ [] = True
   37 eObjetosValido e (h:t) =
   38     if ehDisparo h == True
   39     then
   40         if ePosicaoMapaLivre (posicaoObjeto h) mapa
   41             && not (existeBarril (posicaoObjeto h) t)
   42             && objetoValido h
   43             && disparoValido h
   44             && verificarDonos donos
   45             && donosValido
   46         then eObjetosValido e t
   47         else False
   48     else
   49         if ePosicaoMapaLivre (posicaoObjeto h) mapa
   50             && ePosicaoEstadoLivre (posicaoObjeto h) e{objetosEstado = t}
   51             && objetoValido h
   52             && disparoValido h
   53             && verificarDonos donos
   54             && donosValido
   55         then eObjetosValido e t
   56         else False
   57 
   58                         
   59     where
   60         mapa = mapaEstado e
   61         minhocas = minhocasEstado e -- * MINHOCAS
   62         donos = listaDonos (h:t)
   63 
   64         donosValido = if length donos > length minhocas then False else True
   65 
   66         objetoValido obj = case obj of
   67             Disparo{tipoDisparo = Escavadora} -> False
   68             Disparo{tipoDisparo = Jetpack}  -> False
   69             Disparo{tipoDisparo = Bazuca} -> disparoValido obj
   70             Disparo{tipoDisparo = Mina} -> disparoValido obj
   71             Disparo{tipoDisparo = Dinamite} -> disparoValido obj
   72             _             -> True --todo caso exista novos casos adicionar

   79 -- * Verifica se o disparo é valido
   80 
   81 disparoValido :: Objeto -> Bool
   82 disparoValido d@Disparo{tipoDisparo = Bazuca} = if (tempoDisparo d == Nothing) then True else False
   83 disparoValido d@Disparo{tipoDisparo = Mina} = if (tempoDisparo d <= Just 2 && tempoDisparo d >= Just 0) || (tempoDisparo d == Nothing) then True else False
   84 disparoValido d@Disparo{tipoDisparo = Dinamite} = if (tempoDisparo d <= Just 4 && tempoDisparo d >= Just 0) then True else False
   85 disparoValido _ = True

   87 -- * Retorna uma lista de todos os donos dos disparos no estado
   88 
   89 listaDonos :: [Objeto] -> [Int]
   90 listaDonos [] = []
   91 listaDonos (h:t) = if ehDisparo h
   92                         then [x] ++ listaDonos t 
   93                         else listaDonos t
   94                     where
   95                         x = donoDisparo h

  100 verificarDonos [] = True

  107 eMinhocasValidas :: Estado -> [Minhoca] -> Bool
  108 eMinhocasValidas _ [] = True
  109 eMinhocasValidas e (h:t) = posicaoValida && morteOk && vidaOk && eMinhocasValidas e t
  110   where
  111     posicaoValida = case posicaoMinhoca h of
  112         Nothing   -> True  
  113         Just p    -> ePosicaoEstadoLivre p e{minhocasEstado = t}
  114 
  115     morteOk = validaMorte h (mapaEstado e)
  116     vidaOk  = verificaVida h

  123 -- * Verifica se a minhoca esta dentro de água, e caso esteja, verifica se a mesma esta morta
  124 -- * Retorna True se a minhoca estiver valida (pos valida e nao morta) e False caso esteja em agua e nao esteja morta
  125 
  126 validaMorte :: Minhoca -> Mapa -> Bool
  127 validaMorte minh [] = False
  128 validaMorte minh m = case posicaoMinhoca minh of
  129     Nothing ->
  130         vidaMinhoca minh == Morta
  131 
  132     Just pos -> case terrenoAtual of
  133                                     Agua -> vidaMinhoca minh == Morta
  134                                     outro    -> True
  135                 where
  136                     terrenoAtual = case (encontraPosicaoMatriz pos m) of
  137                                         Just a -> a 
  138                                         Nothing -> Ar -- ! atencao
  139 -- * Verifica se a vida da minhoca é valida
  140 
  141 verificaVida :: Minhoca -> Bool
  142 verificaVida m = case vida of
  143                     Morta -> True
  144                     Viva a -> if (a >= 0 && a <= 100) then True else False
  145     where
  146         vida = vidaMinhoca m
