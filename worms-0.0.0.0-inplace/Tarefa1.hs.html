   17 -- | Função principa*! Excecao nl da Tarefa 1. Recebe um estado e retorna se este é válido ou não.
   18 validaEstado :: Estado -> Bool
   19 validaEstado e = eMapaValido mapa && eObjetosValido e objetos && eMinhocasValidas e minhocas
   20     where
   21         mapa = mapaEstado e
   22         objetos = objetosEstado e
   23         minhocas = minhocasEstado e

   37 eObjetosValido :: Estado -> [Objeto] -> Bool
   38 eObjetosValido _ [] = True
   39 eObjetosValido e (h:t) =
   40     if ehDisparo h == True
   41         then
   42             if not (existeBarril (posicaoObjeto h) t)
   43                 && objetoValido h
   44                 && disparoValido mapa h
   45                 && donoValido e h
   46             then eObjetosValido e t
   47             else False
   48         else
   49             if ePosicaoMapaLivre (posicaoObjeto h) mapa
   50                 && ePosicaoEstadoLivre (posicaoObjeto h) e{objetosEstado = t}
   51             then eObjetosValido e t
   52             else False
   53                         
   54     where
   55         mapa = mapaEstado e
   56         minhocas = minhocasEstado e -- * MINHOCAS
   57         
   58         
   59 
   60         objetoValido obj = case obj of
   61             Disparo{tipoDisparo = Escavadora} -> False
   62             Disparo{tipoDisparo = Jetpack}  -> False
   63             Disparo{tipoDisparo = Bazuca} -> disparoValido mapa obj
   64             Disparo{tipoDisparo = Mina} -> disparoValido mapa obj
   65             Disparo{tipoDisparo = Dinamite} -> disparoValido mapa obj
   66             _ -> True --todo caso exista novos casos adicionar
   67

   73 disparoValido :: Mapa -> Objeto -> Bool
   74 disparoValido m d@Disparo{tipoDisparo = Bazuca} = if (tempoDisparo d == Nothing && disparoBazucaValido d m) then True else False
   75 disparoValido m d@Disparo{tipoDisparo = Mina} = if (tempoDisparo d <= Just 2 && tempoDisparo d >= Just 0 && ePosicaoMapaLivre (posicaoObjeto d) m) || (tempoDisparo d == Nothing && ePosicaoMapaLivre (posicaoObjeto d) m) then True else False
   76 disparoValido m d@Disparo{tipoDisparo = Dinamite} = if (tempoDisparo d <= Just 4 && tempoDisparo d >= Just 0 && ePosicaoMapaLivre (posicaoObjeto d) m) then True else False
   77 disparoValido m _ = True

   81 disparoBazucaValido :: Objeto -> Mapa -> Bool -- * e livre se n tiver opaco
   82 disparoBazucaValido obj mapa = if (ePosicaoMapaLivre prevPos mapa) then True else False
   83                     where
   84                         pos = posicaoObjeto obj
   85                         dir = direcaoDisparo obj
   86                         prevPos = movePosicao (direcaoOposta dir) pos

   96 donoValido :: Estado -> Objeto -> Bool
   97 donoValido e obj = if indiceValido && verificaLista (listaDonos listaDisparos) then True else False
   98             where
   99                 indiceDono = donoDisparo obj
  100                 minhocas = minhocasEstado e
  101                 objetos = objetosEstado e
  102                 listaDisparos = disparosEstado objetos
  103                 -- Primeira verificação (é um indice valido na lista de minhocas)
  104                 indiceValido = if indiceDono <= (length minhocas - 1) && indiceDono >= 0 then True else False
  105 
  106 
  107                 -- * Recebe os objetos do estado e devolve uma lista com (tipo de arma, dono)
  108                 -- Exemplo [(Dinamite, 0,(Mina,0)] -> Valida
  109                 -- Exemplo [(Dinamite, 0),(Dinamite,0)] -> Invalido
  110 
  111                 
  112                 verificaLista :: [(TipoArma, Int)] -> Bool
  113                 verificaLista [] = True
  114                 verificaLista (h:t) = if elem h t then False else True
  115 
  116 
  117                 -- * Devolve os objetos disparos
  118                 disparosEstado :: [Objeto] -> [Objeto]
  119                 disparosEstado [] = []
  120                 disparosEstado (h:t) = if ehDisparo h then h : disparosEstado t else disparosEstado t
  121

  123 listaDonos :: [Objeto] -> [(TipoArma, Int)]
  124 listaDonos [] = []
  125 listaDonos (h:t) = (tipo,indice) : listaDonos t
  126                 where
  127                     indice = donoDisparo h
  128                     tipo = tipoDisparo h

  134 
  135 eMinhocasValidas :: Estado -> [Minhoca] -> Bool
  136 eMinhocasValidas _ [] = True
  137 eMinhocasValidas e (h:t) = posicaoValida && morteOk && vidaOk && eMinhocasValidas e t && armasValidas
  138   where
  139     posicaoValida = case posicaoMinhoca h of
  140         Nothing   -> True  
  141         Just p    -> ePosicaoEstadoLivre p e{minhocasEstado = t}
  142 
  143     morteOk = validaMorte h (mapaEstado e)
  144     vidaOk  = verificaVida h
  145 
  146     armasValidas = if jetpackMinhoca h >= 0 && escavadoraMinhoca h >= 0 && bazucaMinhoca h >= 0 && minaMinhoca h >= 0 && dinamiteMinhoca h >= 0 then True else False

  157 validaMorte :: Minhoca -> Mapa -> Bool
  158 validaMorte minh [] = False
  159 validaMorte minh m = case posicaoMinhoca minh of
  160     Nothing ->
  161         vidaMinhoca minh == Morta
  162 
  163     Just pos -> case terrenoAtual of
  164                                     Agua -> vidaMinhoca minh == Morta
  165                                     outro    -> True
  166                 where
  167                     terrenoAtual = case (encontraPosicaoMatriz pos m) of
  168                                         Just a -> a 
  169                                         Nothing -> Ar -- ! atencao
  170 -- * Verifica se a vida da minhoca é valida
  171 
  172 verificaVida :: Minhoca -> Bool
  173 verificaVida m = case vida of
  174                     Morta -> True
  175                     Viva a -> if (a >= 0 && a <= 100) then True else False
  176     where
  177         vida = vidaMinhoca m
