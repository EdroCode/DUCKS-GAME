   26 -- | Função principa*! Excecao nl da Tarefa 1. Recebe um estado e retorna se este é válido ou não.
   27 validaEstado :: Estado -> Bool
   28 validaEstado e = eMapaValido mapa && eObjetosValido e objetos && eMinhocasValidas e minhocas
   29     where
   30         mapa = mapaEstado e
   31         objetos = objetosEstado e
   32         minhocas = minhocasEstado e
   33

   36 eMapaValido :: Mapa -> Bool 
   37 eMapaValido [] = False    
   38 eMapaValido m = (eMatrizValida m)

   42 -- * Valida se o objeto é valido
   43 
   44 eObjetosValido :: Estado -> [Objeto] -> Bool
   45 eObjetosValido _ [] = True
   46 eObjetosValido e (h:t) = if ePosicaoMapaLivre (posicaoObjeto h) mapa 
   47                         && ePosicaoEstadoLivre (posicaoObjeto h) e{objetosEstado = t} 
   48                         && objetoValido h
   49                         && disparoValido h
   50                         && verificarDonos donos
   51                         then eObjetosValido e t else False
   52     where
   53         mapa = mapaEstado e
   54         minhocas = minhocasEstado e
   55         donos = listaDonos (h:t)
   56      
   57 
   58         objetoValido obj = case obj of
   59             Disparo{tipoDisparo = Escavadora} -> False
   60             Disparo{tipoDisparo = Jetpack}  -> False
   61             Disparo{tipoDisparo = Bazuca} -> disparoValido obj
   62             Disparo{tipoDisparo = Mina} -> disparoValido obj
   63             Disparo{tipoDisparo = Dinamite} -> disparoValido obj
   64             _             -> True --todo caso exista novos casos adicionar

   69 -- * Verifica se o objeto fornecido é um disparo
   70 
   71 ehDisparo :: Objeto -> Bool
   72 ehDisparo d@Disparo{} = True
   73 ehDisparo _ = False
   74 
   75 -- * Verifica se o disparo é valido
   76 
   77 disparoValido :: Objeto -> Bool
   78 disparoValido d@Disparo{tipoDisparo = Bazuca} = if (tempoDisparo d == Nothing) then True else False
   79 disparoValido d@Disparo{tipoDisparo = Mina} = if (tempoDisparo d <= Just 2 && tempoDisparo d >= Just 0) then True else False
   80 disparoValido d@Disparo{tipoDisparo = Dinamite} = if (tempoDisparo d <= Just 4 && tempoDisparo d >= Just 0) then True else False
   81 disparoValido _ = True
   82 
   83 -- * Retorna uma lista de todos os donos dos disparos no estado
   84 
   85 listaDonos :: [Objeto] -> [Int]
   86 listaDonos [] = []
   87 listaDonos (h:t) = if ehDisparo h
   88                         then [x] ++ listaDonos t 
   89                         else listaDonos t
   90                     where
   91                         x = donoDisparo h

   93 -- * Verifica se existe donos repetidos

   95 verificarDonos :: Eq a => [a] -> Bool
   96 verificarDonos [] = True
   97 verificarDonos (h:t) =
   98   if elem h t then False
   99   else verificarDonos t

  106 eMinhocasValidas :: Estado -> [Minhoca] -> Bool
  107 eMinhocasValidas e [] = True
  108 eMinhocasValidas e (h:t) = if ePosicaoEstadoLivre pos e{minhocasEstado = t} 
  109                            && validaMorte h (mapaEstado e)
  110                            && verificaVida h
  111                            then eMinhocasValidas e t else False
  112     where
  113         pos = case posicaoMinhoca h of 
  114             Just (y,x) -> (y,x) 
  115             Nothing -> (-1,-1) -- !  perguntar ao stor
  116         vida = case vidaMinhoca h of Viva a -> a
  117         municoes = [jetpackMinhoca, escavadoraMinhoca, bazucaMinhoca, minaMinhoca, dinamiteMinhoca]
  118         muniValidas :: [Int] -> Bool
  119         muniValidas [] = True
  120         muniValidas (h:t) = if (h >= 0) then muniValidas t else False

  127 -- * Verifica se a minhoca esta dentro de água, e caso esteja, verifica se a mesma esta morta
  128 -- * Retorna True se a minhoca estiver valida (pos valida e nao morta) e False caso esteja em agua e nao esteja morta
  129 validaMorte :: Minhoca -> Mapa -> Bool
  130 validaMorte minh [] = False
  131 validaMorte minh m = if pos == (-1, -1) || terrenoAtual == Agua -- ! mesma situacao
  132                         then if minhocaViva then True else False
  133                         else True                      
  134   where
  135     
  136     vida = case vidaMinhoca minh of
  137         Morta -> 0
  138         Viva a -> a
  139     
  140     pos = case posicaoMinhoca minh of Just (y,x) -> (y,x)
  141     terrenoAtual = case (encontraPosicaoMatriz pos m) of Just a -> a 
  142                                                          Nothing -> Ar
  143     
  144     minhocaViva :: Bool
  145     minhocaViva = if vida == 0 then False else True
  146

  148 -- * Verifica se a vida da minhoca é valida
  149 
  150 verificaVida :: Minhoca -> Bool
  151 verificaVida m = case vida of
  152                     Morta -> True
  153                     Viva a -> if (a >= 0 && a <= 100) then True else False
  154     where
  155         vida = vidaMinhoca m
