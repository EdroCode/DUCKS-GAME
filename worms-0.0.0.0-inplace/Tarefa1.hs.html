   25 -- | Função principa*! Excecao nl da Tarefa 1. Recebe um estado e retorna se este é válido ou não.
   26 validaEstado :: Estado -> Bool
   27 validaEstado e = eMapaValido mapa && eObjetosValido e objetos && eMinhocasValidas e minhocas
   28     where
   29         mapa = mapaEstado e
   30         objetos = objetosEstado e
   31         minhocas = minhocasEstado e

   39 eMapaValido [] = False

   44 -- * Valida se o objeto é valido
   45 
   46 eObjetosValido :: Estado -> [Objeto] -> Bool
   47 eObjetosValido _ [] = True
   48 eObjetosValido e (h:t) = if ePosicaoMapaLivre (posicaoObjeto h) mapa 
   49                         && ePosicaoEstadoLivre (posicaoObjeto h) e{objetosEstado = t} 
   50                         && objetoValido h
   51                         && disparoValido h
   52                         && verificarDonos donos
   53                         && donosValido
   54                         then eObjetosValido e t else False
   55     where
   56         mapa = mapaEstado e
   57         minhocas = minhocasEstado e
   58         donos = listaDonos (h:t)
   59 
   60         donosValido = if length donos > length minhocas then False else True
   61 
   62         objetoValido obj = case obj of
   63             Disparo{tipoDisparo = Escavadora} -> False
   64             Disparo{tipoDisparo = Jetpack}  -> False
   65             Disparo{tipoDisparo = Bazuca} -> disparoValido obj
   66             Disparo{tipoDisparo = Mina} -> disparoValido obj
   67             Disparo{tipoDisparo = Dinamite} -> disparoValido obj
   68             _             -> True --todo caso exista novos casos adicionar
   69

   73 -- * Verifica se o objeto fornecido é um disparo
   74 
   75 ehDisparo :: Objeto -> Bool
   76 ehDisparo d@Disparo{} = True
   77 ehDisparo _ = False
   78 
   79 -- * Verifica se o disparo é valido
   80 
   81 disparoValido :: Objeto -> Bool
   82 disparoValido d@Disparo{tipoDisparo = Bazuca} = if (tempoDisparo d == Nothing) then True else False
   83 disparoValido d@Disparo{tipoDisparo = Mina} = if (tempoDisparo d <= Just 2 && tempoDisparo d >= Just 0) then True else False
   84 disparoValido d@Disparo{tipoDisparo = Dinamite} = if (tempoDisparo d <= Just 4 && tempoDisparo d >= Just 0) then True else False
   85 disparoValido _ = True
   86 
   87 -- * Retorna uma lista de todos os donos dos disparos no estado
   88 
   89 listaDonos :: [Objeto] -> [Int]
   90 listaDonos [] = []
   91 listaDonos (h:t) = if ehDisparo h
   92                         then [x] ++ listaDonos t 
   93                         else listaDonos t
   94                     where
   95                         x = donoDisparo h

  110 
  111 eMinhocasValidas :: Estado -> [Minhoca] -> Bool
  112 eMinhocasValidas _ [] = True
  113 eMinhocasValidas e (h:t) = posicaoValida && morteOk && vidaOk && eMinhocasValidas e t
  114   where
  115     posicaoValida = case posicaoMinhoca h of
  116         Nothing   -> True  
  117         Just p    -> ePosicaoEstadoLivre p e{minhocasEstado = t}
  118 
  119     morteOk = validaMorte h (mapaEstado e)
  120     vidaOk  = verificaVida h

  127 -- * Verifica se a minhoca esta dentro de água, e caso esteja, verifica se a mesma esta morta
  128 -- * Retorna True se a minhoca estiver valida (pos valida e nao morta) e False caso esteja em agua e nao esteja morta
  129 
  130 validaMorte :: Minhoca -> Mapa -> Bool
  131 validaMorte minh [] = False
  132 validaMorte minh m = case posicaoMinhoca minh of
  133     Nothing ->
  134         vidaMinhoca minh == Morta
  135 
  136     Just pos -> case terrenoAtual of
  137                                     Agua -> vidaMinhoca minh == Morta
  138                                     outro    -> True
  139                 where
  140                     terrenoAtual = case (encontraPosicaoMatriz pos m) of
  141                                         Just a -> a 
  142                                         Nothing -> Ar -- ! atencao
  143 -- * Verifica se a vida da minhoca é valida
  144 
  145 verificaVida :: Minhoca -> Bool
  146 verificaVida m = case vida of
  147                     Morta -> True
  148                     Viva a -> if (a >= 0 && a <= 100) then True else False

  150         vida = vidaMinhoca m
