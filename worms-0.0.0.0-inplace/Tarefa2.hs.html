  189 -- Verifica se a posicao que a minhoca ira é nao opaca
  190 eNovaPosLivre :: Posicao -> Estado -> Bool
  191 eNovaPosLivre posNova est = if ePosicaoEstadoLivre posNova est == True then True else False

  194 estaNoSolo :: Posicao -> Mapa -> Bool
  195 estaNoSolo p [] = False
  196 estaNoSolo pos mapa = if eTerrenoOpaco blocoInferior && not (eTerrenoOpaco blocoAtual) then True else False
  197             where
  198                 blocoInferior = case encontraPosicaoMatriz (movePosicao Sul pos) mapa of Just a -> a
